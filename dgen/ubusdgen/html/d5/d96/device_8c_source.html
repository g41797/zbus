154static int set_device_state(struct device *dev, bool state)
155{
156 if (state) {
157 /* Get ifindex for all devices being enabled so a valid */
158 /* ifindex is in place avoiding possible race conditions */
159 device_set_ifindex(dev, system_if_resolve(dev));
160 if (!dev->ifindex)
161 return -1;
162
163 system_if_get_settings(dev, &dev->orig_settings);
164 /* Only keep orig settings based on what needs to be set */
165 dev->orig_settings.valid_flags = dev->orig_settings.flags;
166 dev->orig_settings.flags &= dev->settings.flags;
167 system_if_apply_settings(dev, &dev->settings, dev->settings.flags);
168
169 if (!dev->external)
170 system_if_up(dev);
171
172 system_if_apply_settings_after_up(dev, &dev->settings);
173 } else {
174 if (!dev->external)
175 system_if_down(dev);
176 system_if_apply_settings(dev, &dev->orig_settings, dev->orig_settings.flags);
177
178 /* Restore any settings present in UCI which may have
179 * failed to apply so that they will be re-attempted
180 * the next time the device is brought up */
181 dev->settings.flags |= dev->settings.valid_flags;
182 }
183
184 return 0;
185}

187static int
188simple_device_set_state(struct device *dev, bool state)
189{
190 struct device *pdev;
191 int ret = 0;
192
193 pdev = dev->parent.dev;
194 if (state && !pdev) {
195 pdev = system_if_get_parent(dev);
196 if (pdev)
197 device_add_user(&dev->parent, pdev);
198 }
199
200 if (pdev) {
201 if (state)
202 ret = device_claim(&dev->parent);
203 else
204 device_release(&dev->parent);
205
206 if (ret < 0)
207 return ret;
208 }
209 return set_device_state(dev, state);
210}

212static struct device *
213simple_device_create(const char *name, struct device_type *devtype,
214 struct blob_attr *attr)
215{
216 struct blob_attr *tb[__DEV_ATTR_MAX];
217 struct device *dev = NULL;
218
219 /* device type is unused for simple devices */
220 devtype = NULL;
221
222 blobmsg_parse(dev_attrs, __DEV_ATTR_MAX, tb, blob_data(attr), blob_len(attr));
223 dev = device_get(name, true);
224 if (!dev)
225 return NULL;
226
227 dev->set_state = simple_device_set_state;
228 device_init_settings(dev, tb);
229
230 return dev;
231}

315static bool device_fill_vlan_range(struct device_vlan_range *r, const char *val)
316{
317 unsigned long cur_start, cur_end;
318 char *sep;
319
320 cur_start = strtoul(val, &sep, 0);
321 cur_end = cur_start;
322
323 if (*sep == '-')
324 cur_end = strtoul(sep + 1, &sep, 0);
325 if (*sep || cur_end < cur_start)
326 return false;
327
328 r->start = cur_start;
329 r->end = cur_end;
330
331 return true;
332}

334static void
335device_set_extra_vlans(struct device *dev, struct blob_attr *data)
336{
337 struct blob_attr *cur;
338 int n_vlans;
339 size_t rem;
340
341 dev->n_extra_vlan = 0;
342 if (!data)
343 return;
344
345 n_vlans = blobmsg_check_array(data, BLOBMSG_TYPE_STRING);
346 if (n_vlans < 1)
347 return;
348
349 dev->extra_vlan = realloc(dev->extra_vlan, n_vlans * sizeof(*dev->extra_vlan));
350 blobmsg_for_each_attr(cur, data, rem)
351 if (device_fill_vlan_range(&dev->extra_vlan[dev->n_extra_vlan],
352 blobmsg_get_string(cur)))
353 dev->n_extra_vlan++;
354}

358{

360 struct blob_attr *cur;
361 struct ether_addr *ea;
362 bool disabled = false;
363
364 if (dev->wireless)
365 s->flags &= DEV_OPT_ISOLATE;
366 else
367 s->flags = 0;
368 if ((cur = tb[DEV_ATTR_ENABLED]))
369 disabled = !blobmsg_get_bool(cur);
370
371 if ((cur = tb[DEV_ATTR_MTU]) && blobmsg_get_u32(cur) >= 68) {
372 s->mtu = blobmsg_get_u32(cur);
373 s->flags |= DEV_OPT_MTU;
374 }
375
376 if ((cur = tb[DEV_ATTR_MTU6]) && blobmsg_get_u32(cur) >= 1280) {
377 s->mtu6 = blobmsg_get_u32(cur);
378 s->flags |= DEV_OPT_MTU6;

546 s->asym_pause = blobmsg_get_bool(cur);
547 s->flags |= DEV_OPT_ASYM_PAUSE;
548 }

550 if ((cur = tb[DEV_ATTR_RXPAUSE])) {
551 s->rxpause = blobmsg_get_bool(cur);
552 s->flags |= DEV_OPT_RXPAUSE;
553 }
554
555 if ((cur = tb[DEV_ATTR_TXPAUSE])) {
556 s->txpause = blobmsg_get_bool(cur);
557 s->flags |= DEV_OPT_TXPAUSE;
558 }
559
560 if ((cur = tb[DEV_ATTR_AUTONEG])) {
561 s->autoneg = blobmsg_get_bool(cur);
562 s->flags |= DEV_OPT_AUTONEG;
563 }
564
565 if ((cur = tb[DEV_ATTR_GRO])) {
566 s->gro = blobmsg_get_bool(cur);
567 s->flags |= DEV_OPT_GRO;
568 }
569
570 if ((cur = tb[DEV_ATTR_MASTER])) {
571 char *ifname = blobmsg_get_string(cur);
572 s->master_ifindex = if_nametoindex(ifname);
573 s->flags |= DEV_OPT_MASTER;
574 }
575
576 if ((cur = tb[DEV_ATTR_EEE])) {
577 s->eee = blobmsg_get_bool(cur);
578 s->flags |= DEV_OPT_EEE;
579 }
580
581 /* Remember the settings present in UCI */
582 s->valid_flags = s->flags;
583
584 cur = tb[DEV_ATTR_AUTH_VLAN];
585 free(dev->config_auth_vlans);
586 dev->config_auth_vlans = cur ? blob_memdup(cur) : NULL;

590}

713int device_claim(struct device_user *dep)
714{
715 struct device *dev = dep->dev;
716 int ret = 0;
717
718 if (dep->claimed)
719 return 0;
720
721 if (!dev)
722 return -1;
723
724 dep->claimed = true;
725 D(DEVICE, "Claim %s %s, new active count: %d", dev->type->name, dev->ifname, dev->active + 1);
726 if (++dev->active != 1)
727 return 0;
728
729 device_broadcast_event(dev, DEV_EVENT_SETUP);
730 device_fill_default_settings(dev);
731 ret = dev->set_state(dev, true);
732 if (ret == 0)
733 device_broadcast_event(dev, DEV_EVENT_UP);
734 else {
735 D(DEVICE, "claim %s %s failed: %d", dev->type->name, dev->ifname, ret);
736 dev->active = 0;
737 dep->claimed = false;
738 }
739
740 return ret;
741}

743void device_release(struct device_user *dep)
744{
745 struct device *dev = dep->dev;
746
747 if (!dep->claimed)
748 return;
749
750 dep->claimed = false;
751 dev->active--;
752 D(DEVICE, "Release %s %s, new active count: %d", dev->type->name, dev->ifname, dev->active);
753 assert(dev->active >= 0);
754
755 if (dev->active)
756 return;
757
758 device_broadcast_event(dev, DEV_EVENT_TEARDOWN);
759 dev->set_state(dev, false);
760
761 if (dev->active)
762 return;
763
764 device_broadcast_event(dev, DEV_EVENT_DOWN);
765}

1162static enum dev_change_type
1163device_set_config(struct device *dev, struct device_type *type,
1164 struct blob_attr *attr)
1165{
1166 struct blob_attr *tb[__DEV_ATTR_MAX];
1167 const struct uci_blob_param_list *cfg = type->config_params;
1168
1169 if (type != dev->type)
1170 return DEV_CONFIG_RECREATE;
1171
1172 if (dev->type->reload)
1173 return dev->type->reload(dev, attr);
1174
1175 if (uci_blob_check_equal(dev->config, attr, cfg))
1176 return DEV_CONFIG_NO_CHANGE;
1177
1178 if (cfg == &device_attr_list) {
1179 memset(tb, 0, sizeof(tb));
1180
1181 if (attr)
1182 blobmsg_parse(dev_attrs, __DEV_ATTR_MAX, tb,
1183 blob_data(attr), blob_len(attr));
1184
1185 device_init_settings(dev, tb);
1186 return DEV_CONFIG_RESTART;
1187 } else
1188 return DEV_CONFIG_RECREATE;

1191enum dev_change_type
1192device_apply_config(struct device *dev, struct device_type *type,
1193 struct blob_attr *config)
1194{
1195 enum dev_change_type change;
1196
1197 change = device_set_config(dev, type, config);
1198 switch (change) {
1199 case DEV_CONFIG_RESTART:
1200 case DEV_CONFIG_APPLIED:
1201 D(DEVICE, "Device '%s': config applied", dev->ifname);
1202 config = blob_memdup(config);
1203 free(dev->config);
1204 dev->config = config;
1205 if (change == DEV_CONFIG_RESTART && dev->present) {
1206 int ret = 0;
1207
1208 device_set_present(dev, false);
1209 if (dev->active) {
1210 ret = dev->set_state(dev, false);
1211 if (!ret)
1212 ret = dev->set_state(dev, true);
1213 }
1214 if (!ret)
1215 device_set_present(dev, true);
1216 }
1217 break;
1218 case DEV_CONFIG_NO_CHANGE:
1219 D(DEVICE, "Device '%s': no configuration change", dev->ifname);
1220 break;
1221 case DEV_CONFIG_RECREATE:
1222 break;
1223 }
1224
1225 return change;
1226}
