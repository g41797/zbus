225static inline int system_get_addr_family(unsigned int flags)
226{
227 if ((flags & DEVADDR_FAMILY) == DEVADDR_INET6)
228 return AF_INET6;
229 else
230 return AF_INET;
231}

233static inline int system_get_addr_len(unsigned int flags)
234{
235 if ((flags & DEVADDR_FAMILY) != DEVADDR_INET6)
236 return sizeof(struct in_addr);
237 else
238 return sizeof(struct in6_addr);
239}
