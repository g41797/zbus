62int safe_list_for_each(struct safe_list *head,
63 int (*cb)(void *ctx, struct safe_list *list),
64 void *ctx)
65{

71 __safe_list_set_iterator(cur, &i);
72 cur != head;
73 cur = i.next, __safe_list_move_iterator(cur, &i)) {
74 ret = cb(ctx, cur);

77 }

96{
