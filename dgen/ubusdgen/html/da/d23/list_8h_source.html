184static inline void
185_list_splice(const struct list_head *list, struct list_head *prev,
186 struct list_head *next)
187{
188 struct list_head *first;
189 struct list_head *last;
190
191 if (list_empty(list))
192 return;
193
194 first = list->next;
195 last = list->prev;
196 first->prev = prev;
197 prev->next = first;
198 last->next = next;
199 next->prev = last;
200}
