211static int system_route_msg(struct device *dev, struct device_route *route, const char *type)
212{
213 char addr[64], gw[64] = " gw ", devstr[64] = "";
214 int af = system_get_addr_family(route->flags);
215 int alen = system_get_addr_len(route->flags);
216 static uint32_t zero_addr[4];
217
218 if ((route->flags & DEVADDR_FAMILY) != DEVADDR_INET4)
219 return -1;
220
221 if (!route->mask)
222 sprintf(addr, "default");
223 else
224 inet_ntop(af, &route->addr.in, addr, sizeof(addr));
225
226 if (memcmp(&route->nexthop.in, (void *) zero_addr, alen) != 0)
227 inet_ntop(af, &route->nexthop.in, gw + 4, sizeof(gw) - 4);
228 else
229 gw[0] = 0;
230
231 if (dev)
232 sprintf(devstr, " dev %s", dev->ifname);
233
234 if (route->metric > 0)
235 sprintf(devstr, " metric %d", route->metric);
236
237 D(SYSTEM, "route %s %s%s%s", type, addr, gw, devstr);
238 return 0;
239}

241static int system_neighbor_msg(struct device *dev, struct device_neighbor *neighbor, const char *type)
242{
243 char addr[64];
244 int af = system_get_addr_family(neighbor->flags);
245 inet_ntop(af, &neighbor->addr.in , addr, sizeof(addr));
246
247 D(SYSTEM, "neigh %s %s%s%s %s", type, addr, neighbor->proxy ? "proxy " : "",
248 (neighbor->flags & DEVNEIGH_MAC) ? format_macaddr(neighbor->macaddr) : "",
249 neighbor->router ? "router": "");
250 return 0;
251}
