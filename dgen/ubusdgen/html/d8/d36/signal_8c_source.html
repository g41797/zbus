33static void signal_shutdown(int signal, siginfo_t *siginfo, void *data)
34{
35 int event = 0;
36 char *msg = NULL;
37
38#ifndef DISABLE_INIT
39 switch(signal) {
40 case SIGINT:
41 case SIGTERM:
42 event = RB_AUTOBOOT;
43 msg = "reboot";
44 break;
45 case SIGUSR1:
46 case SIGUSR2:
47 case SIGPWR:
48 event = RB_POWER_OFF;
49 msg = "poweroff";
50 break;
51 }
52#endif
53
54 DEBUG(1, "Triggering %s\n", msg);
55 if (event)
56 procd_shutdown(event);
57}

85void procd_signal(void)
86{
87 signal(SIGPIPE, SIG_IGN);
88 if (getpid() != 1)
89 return;
90 sigaction(SIGTERM, &sa_shutdown, NULL);
91 sigaction(SIGINT, &sa_shutdown, NULL);
92 sigaction(SIGUSR1, &sa_shutdown, NULL);
93 sigaction(SIGUSR2, &sa_shutdown, NULL);
94 sigaction(SIGPWR, &sa_shutdown, NULL);
95 sigaction(SIGSEGV, &sa_crash, NULL);
96 sigaction(SIGBUS, &sa_crash, NULL);
97 sigaction(SIGHUP, &sa_dummy, NULL);
98 sigaction(SIGKILL, &sa_dummy, NULL);
99 sigaction(SIGSTOP, &sa_dummy, NULL);
100#ifndef DISABLE_INIT
101 reboot(RB_DISABLE_CAD);
102#endif
103}
